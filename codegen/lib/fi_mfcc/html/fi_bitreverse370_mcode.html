43function x = fi_bitreverse(x,n0) %#codegen
44    n = int32(n0);
45    nv2 = bitsra(n,1);
46    j = int32(1);
47    for i=1:(n-1)
48      if i<j
49        temp = x(j);
50        x(j) = x(i);
51        x(i) = temp;
52      end
53      k = nv2;
54      while k<j
55        j = j-k;
56        k = bitsra(k,1);
57      end
58      j = j+k;
59    end
